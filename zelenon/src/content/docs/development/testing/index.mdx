# Theme Testing & Debugging Guide

## Testing Framework Setup

### Jest Configuration

Set up Jest for theme testing:

```javascript
// jest.config.js
module.exports = {
  testEnvironment: 'jsdom',
  setupFilesAfterEnv: ['<rootDir>/tests/setup.js'],
  moduleNameMapper: {
    '^@/(.*)$': '<rootDir>/src/$1',
    '\\.(css|less|scss|sass)$': 'identity-obj-proxy'
  },
  transform: {
    '^.+\\.js$': 'babel-jest'
  }
};

// tests/setup.js
import '@testing-library/jest-dom';

global.Shopify = {
  shop: 'test-store.myshopify.com',
  currency: {
    active: 'USD',
    rate: '1.0'
  }
};
```

### Unit Testing Components

Create tests for theme components:

```javascript
// tests/components/product-card.test.js
import { ProductCard } from '../components/product-card';

describe('ProductCard', () => {
  const mockProduct = {
    id: 1234567890,
    title: 'Test Product',
    price: 1999,
    compare_at_price: 2999,
    featured_image: {
      src: 'test-image.jpg',
      alt: 'Test Product Image'
    }
  };

  beforeEach(() => {
    document.body.innerHTML = `
      <div class="product-card" data-product-id="${mockProduct.id}">
        <img class="product-card__image" src="" alt="">
        <h3 class="product-card__title"></h3>
        <div class="product-card__price"></div>
      </div>
    `;
  });

  test('renders product information correctly', () => {
    const card = new ProductCard(document.querySelector('.product-card'));
    card.render(mockProduct);

    expect(document.querySelector('.product-card__title')).toHaveTextContent(mockProduct.title);
    expect(document.querySelector('.product-card__price')).toHaveTextContent('$19.99');
  });

  test('shows sale badge when product is on sale', () => {
    const card = new ProductCard(document.querySelector('.product-card'));
    card.render(mockProduct);

    expect(document.querySelector('.product-card__sale-badge')).toBeInTheDocument();
  });
});
```

## Automated Testing

### Integration Tests

Test cart functionality:

```javascript
// tests/integration/cart.test.js
import { Cart } from '../modules/cart';
import { fetchMock } from 'jest-fetch-mock';

describe('Cart Integration', () => {
  let cart;

  beforeEach(() => {
    fetchMock.resetMocks();
    cart = new Cart();
  });

  test('adds item to cart successfully', async () => {
    const mockItem = {
      id: 12345,
      quantity: 1,
      properties: {}
    };

    fetchMock.mockResponseOnce(JSON.stringify({
      items: [mockItem],
      item_count: 1,
      total_price: 1999
    }));

    const result = await cart.addItem(mockItem);

    expect(result.success).toBe(true);
    expect(fetchMock).toHaveBeenCalledWith('/cart/add.js', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        items: [mockItem]
      })
    });
  });

  test('handles cart errors appropriately', async () => {
    fetchMock.mockRejectOnce(new Error('Network error'));

    const result = await cart.addItem({
      id: 12345,
      quantity: 1
    });

    expect(result.success).toBe(false);
    expect(result.error).toBeTruthy();
  });
});
```

## Debugging Tools

### Custom Debug Utility

Create a debugging utility:

```javascript
// utils/debug.js
export class ThemeDebugger {
  constructor(options = {}) {
    this.enabled = options.enabled || false;
    this.level = options.level || 'info';
    this.init();
  }

  init() {
    if (this.enabled) {
      this.injectDebugPanel();
      this.setupEventListeners();
    }
  }

  injectDebugPanel() {
    const panel = document.createElement('div');
    panel.className = 'theme-debug-panel';
    panel.innerHTML = `
      <div class="theme-debug-panel__header">
        Theme Debugger
        <button class="theme-debug-panel__close">Ã—</button>
      </div>
      <div class="theme-debug-panel__content">
        <div class="theme-debug-panel__section">
          <h3>Events</h3>
          <div class="theme-debug-panel__events"></div>
        </div>
        <div class="theme-debug-panel__section">
          <h3>State</h3>
          <div class="theme-debug-panel__state"></div>
        </div>
      </div>
    `;
    document.body.appendChild(panel);
  }

  log(message, type = 'info') {
    if (!this.enabled) return;

    const event = {
      timestamp: new Date().toISOString(),
      message,
      type
    };

    this.logToPanel(event);
    console[type](message);
  }

  logToPanel(event) {
    const eventsContainer = document.querySelector('.theme-debug-panel__events');
    const eventElement = document.createElement('div');
    eventElement.className = `theme-debug-event theme-debug-event--${event.type}`;
    eventElement.innerHTML = `
      <span class="theme-debug-event__time">${event.timestamp}</span>
      <span class="theme-debug-event__message">${event.message}</span>
    `;
    eventsContainer.insertBefore(eventElement, eventsContainer.firstChild);
  }

  setupEventListeners() {
    document.addEventListener('shopify:section:load', (event) => {
      this.log(`Section loaded: ${event.target.dataset.sectionId}`, 'info');
    });

    document.addEventListener('shopify:section:unload', (event) => {
      this.log(`Section unloaded: ${event.target.dataset.sectionId}`, 'info');
    });

    window.addEventListener('error', (event) => {
      this.log(`Error: ${event.message}`, 'error');
    });
  }
}

// Usage
const debugger = new ThemeDebugger({
  enabled: process.env.NODE_ENV === 'development'
});
```

### Performance Monitoring

Create performance tracking tools:

```javascript
// utils/performance.js
export class PerformanceMonitor {
  constructor() {
    this.metrics = new Map();
    this.observers = new Map();
    this.init();
  }

  init() {
    this.observeWebVitals();
    this.observeNetworkRequests();
  }

  observeWebVitals() {
    const observer = new PerformanceObserver((list) => {
      for (const entry of list.getEntries()) {
        this.metrics.set(entry.name, entry.value);
        this.logMetric(entry.name, entry.value);
      }
    });

    observer.observe({ entryTypes: ['web-vitals'] });
    this.observers.set('web-vitals', observer);
  }

  observeNetworkRequests() {
    const observer = new PerformanceObserver((list) => {
      for (const entry of list.getEntries()) {
        if (entry.initiatorType === 'fetch' || entry.initiatorType === 'xmlhttprequest') {
          this.logNetworkRequest(entry);
        }
      }
    });

    observer.observe({ entryTypes: ['resource'] });
    this.observers.set('network', observer);
  }

  logMetric(name, value) {
    console.info(`Performance Metric - ${name}: ${value}`);
  }

  logNetworkRequest(entry) {
    console.info(`Network Request - ${entry.name}:`, {
      duration: entry.duration,
      startTime: entry.startTime,
      transferSize: entry.transferSize
    });
  }

  getMetrics() {
    return Object.fromEntries(this.metrics);
  }

  disconnect() {
    for (const observer of this.observers.values()) {
      observer.disconnect();
    }
  }
}
```

## Best Practices

### Error Handling

Implement proper error boundaries:

```javascript
// utils/error-boundary.js
export class ErrorBoundary {
  constructor(component) {
    this.component = component;
    this.errorState = false;
    this.originalRender = component.render;
    this.wrapRender();
  }

  wrapRender() {
    this.component.render = (...args) => {
      try {
        if (this.errorState) {
          return this.renderErrorState();
        }
        return this.originalRender.apply(this.component, args);
      } catch (error) {
        this.handleError(error);
        return this.renderErrorState();
      }
    };
  }

  handleError(error) {
    this.errorState = true;
    console.error('Component Error:', error);
    
    if (process.env.NODE_ENV === 'development') {
      this.logErrorDetails(error);
    }
  }

  renderErrorState() {
    return `
      <div class="component-error">
        <p>Something went wrong. Please try again later.</p>
        ${process.env.NODE_ENV === 'development' ? 
          '<p>Check the console for more details.</p>' : 
          ''}
      </div>
    `;
  }

  logErrorDetails(error) {
    console.group('Component Error Details');
    console.error(error);
    console.info('Component:', this.component);
    console.info('State:', this.component.state);
    console.groupEnd();
  }
}
```
