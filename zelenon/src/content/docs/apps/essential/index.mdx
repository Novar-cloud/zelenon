### Social Media Integration

#### Instagram Feed Setup

```liquid
{% comment %}sections/instagram-feed.liquid{% endcomment %}
{% schema %}
{
  "name": "Instagram Feed",
  "settings": [
    {
      "type": "text",
      "id": "instagram_token",
      "label": "Instagram Access Token"
    },
    {
      "type": "range",
      "id": "post_count",
      "min": 4,
      "max": 12,
      "step": 2,
      "label": "Number of Posts",
      "default": 6
    }
  ],
  "presets": [
    {
      "name": "Instagram Feed",
      "category": "Social Media"
    }
  ]
}
{% endschema %}

<div class="instagram-feed" data-instagram-feed>
  <h2>{{ section.settings.heading }}</h2>
  <div class="instagram-grid" data-instagram-grid></div>
</div>

<script>
class InstagramFeed {
  constructor(container) {
    this.container = container;
    this.grid = container.querySelector('[data-instagram-grid]');
    this.token = {{ section.settings.instagram_token | json }};
    this.postCount = {{ section.settings.post_count | json }};
    this.init();
  }

  async init() {
    try {
      const posts = await this.fetchPosts();
      this.renderPosts(posts);
    } catch (error) {
      console.error('Error loading Instagram feed:', error);
    }
  }

  async fetchPosts() {
    const response = await fetch(
      `https://graph.instagram.com/me/media?fields=id,caption,media_type,media_url,permalink,thumbnail_url&access_token=${this.token}&limit=${this.postCount}`
    );
    const data = await response.json();
    return data.data;
  }

  renderPosts(posts) {
    this.grid.innerHTML = posts.map(post => `
      <a href="${post.permalink}" class="instagram-item" target="_blank" rel="noopener">
        <img src="${post.media_url}" alt="${post.caption || 'Instagram post'}" loading="lazy">
      </a>
    `).join('');
  }
}

customElements.define('instagram-feed', class extends HTMLElement {
  connectedCallback() {
    new InstagramFeed(this);
  }
});
</script>
```

## Conversion Optimization

### Exit Intent Popup

```javascript
// modules/exit-intent.js
export class ExitIntent {
  constructor(options = {}) {
    this.options = {
      delay: 5000,
      cookieExpiry: 7,
      ...options
    };
    
    this.init();
  }

  init() {
    if (this.hasSeenPopup()) return;
    
    setTimeout(() => {
      this.bindEvents();
    }, this.options.delay);
  }

  bindEvents() {
    document.addEventListener('mouseleave', (e) => {
      if (e.clientY < 0 && !this.hasSeenPopup()) {
        this.showPopup();
      }
    });
  }

  showPopup() {
    const popup = document.createElement('div');
    popup.className = 'exit-popup';
    popup.innerHTML = `
      <div class="exit-popup__content">
        <h2>Wait! Don't leave empty-handed</h2>
        <p>Get 10% off your first order when you sign up for our newsletter</p>
        <form class="exit-popup__form" data-klaviyo-signup>
          <input type="email" placeholder="Enter your email" required>
          <button type="submit">Get Discount</button>
        </form>
        <button class="exit-popup__close">&times;</button>
      </div>
    `;

    document.body.appendChild(popup);
    this.setPopupSeen();
    this.bindPopupEvents(popup);
  }

  bindPopupEvents(popup) {
    popup.querySelector('.exit-popup__close').addEventListener('click', () => {
      popup.remove();
    });

    popup.querySelector('form').addEventListener('submit', (e) => {
      e.preventDefault();
      this.handleSignup(e.target);
    });
  }

  async handleSignup(form) {
    const email = form.querySelector('input[type="email"]').value;
    
    try {
      const response = await fetch('/signup', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({ email })
      });

      if (response.ok) {
        this.showSuccess();
      }
    } catch (error) {
      console.error('Signup failed:', error);
    }
  }

  hasSeenPopup() {
    return document.cookie.includes('exitPopupSeen=true');
  }

  setPopupSeen() {
    const date = new Date();
    date.setDate(date.getDate() + this.options.cookieExpiry);
    document.cookie = `exitPopupSeen=true; expires=${date.toUTCString()}; path=/`;
  }

  showSuccess() {
    const popup = document.querySelector('.exit-popup');
    if (popup) {
      popup.querySelector('.exit-popup__content').innerHTML = `
        <h2>Thank You!</h2>
        <p>Your discount code has been sent to your email.</p>
      `;
      
      setTimeout(() => {
        popup.remove();
      }, 3000);
    }
  }
}
```
